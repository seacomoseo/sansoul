{{- $title := or .title .hanchor .menu_label -}} {{- if $title -}} {{- $hanchor := or .hanchor .menu_label -}} {{- $anchor2 := "" -}} {{- /* NODE AND HS */ -}} {{- $hl := partial "func/hl" (merge . (dict "is_title" true)) -}} {{- $hn := $hl.node -}} {{- $hs := $hl.class -}} {{- /* LINK */ -}} {{- $url := "" -}} {{- if .is_title_url -}} {{- $url = .url -}} {{- $anchor2 = $hanchor -}} {{- $hanchor = .anchor | default "auto" -}} {{- else -}} {{- $header_level := cond .is_modal 3 2 -}} {{- if and .zero (le .level $header_level) -}} {{- $url = cond .is_header page.RelPermalink (print "#" .section.id) -}} {{- end -}} {{- end -}} {{- /* TITLE CLASS */ -}} {{- $line := cond (and (not $url) .line) "line" "" -}} {{- $caps := cond (not .caps | not) "caps" "" -}} {{- $hide := cond (not .title) "hide" "" -}} {{- if not .title -}} {{- $line = "" -}} {{- $caps = "" -}} {{- $hs = "" -}} {{- end -}} {{- $class := slice "box__title" $line $caps $hide $hs .class -}} {{- /* TITLE STYLE */ -}} {{- $style := $hl.style -}} {{- /* TITLE MORE */ -}} {{- $more := "" -}} {{- if and $hanchor (ne $title $hanchor) (not $url) -}} {{- $more = $title | markdownify | plainify -}} {{- $title = print "" $hanchor "" -}} {{- end -}} {{- /* TITLE JOIN */ -}} {{- $attrs := dict "class" $class "style" $style "data-more" $more -}} {{- $attrs = partial "func/attrs" $attrs -}} {{- /* HTML */}} {{ print "<" $hn $attrs ">" | safeHTML -}} {{- /* ICON SIDE */ -}} {{- if eq .icon_pos "side" -}} {{- partial "boxes/icon" . -}} {{- end -}} {{- /* ANCHOR */ -}} {{- if $url -}} {{- $anchor_class := slice (cond (not .is_title_url) "" "box__link") (cond (not .line) "" "line") -}} {{- if not .title -}} {{- $anchor_class = "" -}} {{- end -}} {{- /* with wrap */ -}} {{- partial "link" (dict "class" $anchor_class "label" ($title | markdownify) "anchor" ($hanchor | markdownify) "anchor2" ($anchor2 | markdownify) "url" $url "lock" .lock "nomore" $more ) -}} {{- else -}} {{- /* without wrap */ -}} {{- $title | markdownify -}} {{- end -}} {{- /* END */ -}} {{- print "" | safeHTML -}} {{- end -}}
