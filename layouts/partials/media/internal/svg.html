{{- $file := .src | replaceRE `#.*$` "" -}} {{- $src := resources.Get $file -}} {{- $min := $src -}} {{- if $src -}} {{- $min = $src | minify -}} {{- $alt := .alt -}} {{- if not $alt -}} {{- $alt = $file | replaceRE `^/media/(.+)\..*$` `$1` | humanize -}} {{- end -}} {{- $svg := $min.Content | replaceRE `\n` "" -}} {{- $svg_head := index ($svg | findRE `^`) 0 -}} {{- $svg_width := $svg_head | findRE ` width="(.+?)"` -}} {{- $svg_height := $svg_head | findRE ` height="(.+?)"` -}} {{- $svg_not_size := false -}} {{- if $svg_width | and $svg_height -}} {{- $svg_width = index $svg_width 0 | replaceRE ` width="|"` "" -}} {{- $svg_height = index $svg_height 0 | replaceRE ` height="|"` "" -}} {{- else -}} {{- $view_box := $svg_head | replaceRE `.*?viewBox="([\d\.-]+) ([\d\.-]+) ([\d\.-]+) ([\d\.-]+)".*` `$1,$2,$3,$4` -}} {{- $view_box_split := split $view_box "," -}} {{- $svg_width = add (float (index $view_box_split 2)) (float (index $view_box_split 0)) | int -}} {{- $svg_height = add (float (index $view_box_split 3)) (float (index $view_box_split 1)) | int -}} {{- $svg_not_size = true -}} {{- end -}} {{- if $svg | findRE `svg-src|animat|href||` | or (in $src "#src") -}} {{- if .svg_src | or ($svg | findRE `svg-src`) | or (in $src "#src") -}} {{- $svg_minify := $min.Content -}} {{- if $svg_not_size -}} {{- $svg_minify = replace $svg_minify " {{- else -}} {{- $svg = readFile (print "/assets" $file) -}} {{- if not ($svg_head | findRE ` viewBox=`) -}} {{- $svg = replace $svg " {{- end -}} {{- /* schema */}} {{ if not .notschema -}} {{- partial "schema/image" (dict "url" $min.Permalink "width" $svg_width "height" $svg_height "caption" $alt ) -}} {{- end -}} {{- end -}}
